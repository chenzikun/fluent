[文档]    def start(self, q):
        """控制流程， 从队列中获取数据，并处理"""
        self._start()
        self.__q = q
        while True:
            v = q.get()
            if v == FluentSignal.QueueStopBlockedSignal:
                break
            if not self.is_flex_thread_alive():
                break
            elif v is not None:
                # 开始退出
                begin = datetime.datetime.now()
                with opentracing.tracer.start_active_span(FlowType.output, child_of=v.extra().span):
                    self._handle_item(v.value)
                self.report(begin)
            self._interval()
        self._end()
