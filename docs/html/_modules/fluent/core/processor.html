        """ 如果是一对多关系，则需要重写函数.

        因为从队列中获取到数据涉及到锁的问题，因此不在put函数中获取数据

        Args:
            v: 从input队列中获取到的数据
            out_q (queue.Queue): , 产出队列
        """
        # begin = datetime.datetime.now()
        # r = self._handle_item(v)
        # self.report(begin)
        # out_q.put(r)
        with opentracing.tracer.start_active_span(FlowType.process, child_of=flow_item.extra().span) as scope:
            scope.span.log_kv({'value': flow_item.value})
            begin = datetime.datetime.now()
            r = self._handle_item(flow_item.value)
            if isinstance(r, types.GeneratorType):
                for g_item in r:
                    if g_item is not None:
                        self.item_wrapper(out_q, g_item, scope.span)
                        self.report(begin)
                        begin = datetime.datetime.now()
            else:
                self.report(begin)
                if r is not None:
                    self.item_wrapper(out_q, r, scope.span)

[文档]    def start(self, int_q, out_q):
        """"""
        self._start()
        self.__q = int_q
        while True:
            if not self.is_flex_thread_alive():
                break
            v = int_q.get()
            if v == FluentSignal.QueueStopBlockedSignal:
                break
            else:
                self.put(v, out_q)
        self._end()
